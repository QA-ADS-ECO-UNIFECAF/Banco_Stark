        "        \"url\": \"https://swapi.dev/api/starships/10/\"\n",
        "    }\n",
        "}\n",
        "\n",
        "class RespostaLocal:\n",
        "    def __init__(self, status_code, data):\n",
        "        self.status_code = status_code\n",
        "        self._data = data\n",
        "        self.text = json.dumps(data, ensure_ascii=False)\n",
        "\n",
        "    def json(self):\n",
        "        return self._data\n",
        "\n",
        "def buscar_endpoint(url):\n",
        "    try:\n",
        "        resposta = requests.get(url, timeout=10)\n",
        "        if resposta.status_code == 200:\n",
        "            return resposta, \"online\"\n",
        "    except Exception:\n",
        "        pass\n",
        "\n",
        "    if url in DADOS_FALLBACK:\n",
        "        return RespostaLocal(200, DADOS_FALLBACK[url]), \"fallback local\"\n",
        "\n",
        "    return RespostaLocal(404, {\"detail\": \"Not found\"}), \"fallback local\"\n",
        "\n",
        "print(\"Ambiente preparado com sucesso.\")"
      ]
